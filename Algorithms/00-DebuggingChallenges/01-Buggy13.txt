3 - PrintMaxOfArray(arr)
Print the largest element in a given array.

function printMaxOfArray(arr) {
    if (arr.length == 0) {
        console.log("[], no max val.");     could directly return string with input param(arr)
        return;
    }
    var max = 0;
    for (var idx = 0; idx < arr.length; idx++) {
        if (arr[idx] > max){ 
            max = arr[idx]; 
        }
    }
    console.log("Max val is:", max);
}
